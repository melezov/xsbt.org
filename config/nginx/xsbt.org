server {
  listen 80;
  server_name .xsbt.org;

  access_log /var/www/xsbt.org/logs/nginx/access.log;
  error_log /var/www/xsbt.org/logs/nginx/error.log;
  
  location ~ ^/(favicon\.ico|robots\.txt|static/.*)$ {
    root    /var/www/xsbt.org/code/lift/src/main/webapp/;
    expires 30d;
  }

  location / {
    proxy_pass http://localhost:8120;
    proxy_set_header Host $host;
    proxy_set_header X-Forwarded-For $remote_addr;
    proxy_set_header X-Forwarded-Proto http; 
  }
}

server {
  listen 443;
  server_name .xsbt.org;

  ssl on;
  ssl_certificate     /etc/ssl/certs/xsbt.org.pem;
  ssl_certificate_key /etc/ssl/private/xsbt.org.key;

  keepalive_timeout   70;
  ssl_session_timeout 5m;

  ssl_protocols             SSLv3 TLSv1;
  ssl_ciphers               ALL:!ADH:!EXPORT:!SSLv2:RC4+RSA:+HIGH:+MEDIUM;
  ssl_prefer_server_ciphers on; 
  
  access_log /var/www/xsbt.org/logs/nginx/access.log;
  error_log /var/www/xsbt.org/logs/nginx/error.log;
  
  location ~ ^/(favicon\.ico|robots\.txt|static/.*)$ {
    root    /var/www/xsbt.org/code/lift/src/main/webapp/;
    expires 30d;
  }

  location / {
    proxy_pass http://localhost:8120;
    proxy_set_header Host $host;
    proxy_set_header X-Forwarded-For $remote_addr;
    proxy_set_header X-Forwarded-Proto http; 
  }
}
